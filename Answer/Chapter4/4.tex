        \[
            \exists\ (a,b)\in (R_2 \cap R_3),\ (b,c)\in R_1 .    
        \]
        故
        \[
            \begin{cases}
                (a,b) & \in R_2 \\
                (a,b) & \in R_3
            \end{cases}

            \begin{cases}
                (a,c) & \in R_1 \circ R_2\\
                (a,c) & \in R_1 \circ R_3
            \end{cases}

        \]
        即证$R_1 \circ (R_2 \cap R_3) \subseteq (R_1 \circ R_2) \cap (R_1 \circ R_3)$.

\end{proof}
